function switchTab(tab){

  activeDetailTab = tab;

  document.querySelectorAll('.dtab-pane').forEach(p=>p.classList.remove('active'));

}

  const { data } = await sb.from('courses').select('name, schedule_days, schedule_time, status').eq('status','active');

  const canEdit = can('schedule','e');

  const times = ['09:00','11:00','13:00','15:00','18:00','19:00'];

  const days = ['Пн','Вт','Ср','Чт','Пт'];


  const events = [];

  (data||[]).forEach((c,ci)=>{

    (c.schedule_days||[]).forEach(day=>{

      const di = days.indexOf(day);

      const ti = times.indexOf(c.schedule_time);

      if(di>=0 && ti>=0) events.push({di,ti,name:c.name.split(' ').slice(0,2).join(' '),ci});

    });

  });


  let grid = `<div class="schedule-grid">`;

  grid += `<div class="sg-h"></div>`;

  days.forEach(d=>{ grid+=`<div class="sg-h">${d}</div>`; });

  times.forEach((t,ti)=>{

    const last = ti===times.length-1;

    grid += `<div class="sg-t${last?' sg-last':''}">${t}</div>`;

    days.forEach((_,di)=>{

      const ev = events.find(e=>e.di===di&&e.ti===ti);

      const [bg,tc] = ev ? SCHED_COLORS[ev.ci%SCHED_COLORS.length].split('|') : ['',''];

      grid += `<div class="sg-c${last?' sg-last':''}">${ev?`<div class="sg-event" style="background:${bg};color:${tc};"><div class="en">${ev.name}</div></div>`:''}</div>`;

    });

  });

  grid += '</div>';


  el.innerHTML = `

    <div class="sec-top">

      <div class="sec-title">Розклад</div>

      ${canEdit?'<button class="btn-primary" onclick="openCourseModal()">+ Новий курс</button>':''}

    </div>

    ${!canEdit?'<div class="notice">Тільки перегляд. Для змін зверніться до адміна.</div>':''}

    ${grid}`;

}

function toggleCourseField(role){

  const f = document.getElementById('course-field');

  if(f) f.style.display = role==='mentor'?'block':'none';

}


async function saveUser(){

  const id       = document.getElementById('u-id').value.trim();

  const name     = document.getElementById('u-name').value || 'Користувач';

  const role     = document.getElementById('u-role').value;

  const courseId = role==='mentor' ? parseInt(document.getElementById('u-course')?.value)||null : null;

  if(!id){ alert('Вкажи UUID'); return; }

  const { error } = await sb.from('users').upsert([{ id, name, role, course_id: courseId }]);

  if(error){ alert('Помилка: '+error.message); return; }

  closeModal();

  renderUsers();

}


function openEditUserModal(uid){

  const u = allUsers.find(x=>x.id===uid);

  showModal(`<div class="modal">

    <div class="modal-title">Змінити роль — ${u.name}</div>

    <div class="form-row"><label class="form-label">Роль</label>

      <select class="form-select" id="er-role">

        <option value="admin" ${u.role==='admin'?'selected':''}>Адмін</option>

        <option value="mentor" ${u.role==='mentor'?'selected':''}>Ментор</option>

        <option value="reception" ${u.role==='reception'?'selected':''}>Ресепшн</option>

      </select></div>

    <div class="modal-actions">

      <button class="btn-ghost" onclick="closeModal()">Скасувати</button>

      <button class="btn-primary" onclick="applyRole('${uid}')">Зберегти</button>

    </div>

  </div>`);

}


async function applyRole(uid){

  await sb.from('users').update({ role: document.getElementById('er-role').value }).eq('id', uid);

  closeModal();

  renderUsers();

}
